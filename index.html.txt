    <script>
        let cartaAbierta = false;

        // Recuperar parámetros de URL opcionales para personalizar dinámicamente si se desea
        window.addEventListener('DOMContentLoaded', () => {
            const params = new URLSearchParams(window.location.search);
            const para = params.get('para') || 'querida amorcito';
            const de = params.get('de') || 'Tu amorcito que te ama muchotototototote ❤️';

            // Ajustar textos para que entren armoniosamente en el celular
            document.getElementById('label-para-sobre').innerText = para;
            document.getElementById('preview-para').innerText = `Hola ${para},`;
            document.getElementById('modal-para').innerText = `${para}:`;
            document.getElementById('modal-de').innerText = de;
        });

        // Alternar el estado del sobre al pulsar
        function toggleEnvelope() {
            const envelope = document.getElementById('envelope-element');
            
            if (!cartaAbierta) {
                envelope.classList.add('open');
                cartaAbierta = true;
                document.getElementById('instruction-tap').classList.add('invisible');

                // Ejecutar sonidos armónicos y efectos de partículas
                playHarmonicChime();
                triggerFloatingHearts();

                // Mostrar la carta ampliada tras finalizar el despliegue hacia arriba (1.4 segundos)
                setTimeout(() => {
                    if (cartaAbierta) {
                        document.getElementById('modal-screen').classList.remove('hidden');
                    }
                }, 1400);
            }
        }

        // Volver a colocar la carta dentro del sobre
        function cerrarCartaCompleta() {
            document.getElementById('modal-screen').classList.add('hidden');
            const envelope = document.getElementById('envelope-element');
            envelope.classList.remove('open');
            cartaAbierta = false;
            document.getElementById('instruction-tap').classList.remove('invisible');
        }

        // Sintetizador acústico armónico
        function playHarmonicChime() {
            try {
                const AudioContext = window.AudioContext || window.webkitAudioContext;
                const ctx = new AudioContext();
                if (ctx.state === 'suspended') {
                    ctx.resume();
                }
                const now = ctx.currentTime;
                
                const playNote = (frequency, delay, duration) => {
                    const osc = ctx.createOscillator();
                    const gain = ctx.createGain();
                    
                    osc.type = 'sine';
                    osc.frequency.setValueAtTime(frequency, now + delay);
                    
                    gain.gain.setValueAtTime(0, now + delay);
                    gain.gain.linearRampToValueAtTime(0.15, now + delay + 0.04);
                    gain.gain.exponentialRampToValueAtTime(0.0001, now + delay + duration);
                    
                    osc.connect(gain);
                    gain.connect(ctx.destination);
                    
                    osc.start(now + delay);
                    osc.stop(now + delay + duration);
                };

                // Acorde de campanas celestiales brillantes
                playNote(329.63, 0.0, 0.5);  // Mi4
                playNote(392.00, 0.1, 0.5);  // Sol4
                playNote(523.25, 0.2, 0.6);  // Do5
                playNote(659.25, 0.3, 0.8);  // Mi5
                playNote(783.99, 0.4, 1.0);  // Sol5
            } catch (e) {
                console.log("Interacción de audio bloqueada.");
            }
        }

        // Generador de partículas de corazones y destellos flotantes
        function triggerFloatingHearts() {
            const envelopeWrapper = document.querySelector('.envelope-wrapper');
            const emojis = ['❤️', '✨', '💖', '⭐', '💕', '🌸'];
            const maxParticles = 16;
            
            for (let i = 0; i < maxParticles; i++) {
                const particle = document.createElement('div');
                particle.className = 'heart-particle text-lg md:text-2xl';
                particle.innerText = emojis[Math.floor(Math.random() * emojis.length)];
                
                // Distribución horizontal bien posicionada sobre la boca superior del sobre
                const leftOffset = 10 + Math.random() * 80;
                particle.style.left = `${leftOffset}%`;
                particle.style.bottom = `60%`;
                
                const animDelay = Math.random() * 0.4;
                const animDuration = 2.0 + Math.random() * 1.5;
                
                particle.style.animationDelay = `${animDelay}s`;
                particle.style.animationDuration = `${animDuration}s`;
                
                envelopeWrapper.appendChild(particle);
                
                // Remover para conservar la memoria del teléfono
                setTimeout(() => {
                    particle.remove();
                }, (animDelay + animDuration) * 1000);
            }
        }
